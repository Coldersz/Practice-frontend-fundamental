const path = require("path");
const HtmlWebpackPlugin = require("html-webpack-plugin");

module.exports = {
  entry: "./src/index.js",
  output: {
    path: path.resolve(__dirname, "dist"),
    filename: "bundle.js"
  },
  mode: "production",
  module: {
    rules: [
      {
        test: /\.css$/,
        use: [
          {
            loader: "style-loader"
          },
          {
            loader: "css-loader"
          }
        ]
      },
      /* babel-loader */
      {
        test: /\.js$/,
        // properti exclude berarti kita mengecualikan webpack untuk memproses berkas .js yang berada pada folder “node_modules”.
        // dengan tujuan mempercepat proses build pada projek kita
        exclude: "/node_modules/",
        use: [
          {
            loader: "babel-loader",
            options: {
              presets: ["@babel/preset-env"]
            }
          }
        ]
      }
    ]
  },
  plugins: [
    new HtmlWebpackPlugin ({
      template: "./src/template.html", // Template merupakan berkas rujukan bagi pembuatan berkas HTML yang dihasilkan HtmlWebpackPlugin.
      filename: "index.html"    // filename akan digunakan sebagai penamaan berkas HTML yang akan dihasilkan nanti.
    })
  ]
}
